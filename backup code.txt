def export_to_jrxml(df, report_name="report"):
    import xml.etree.ElementTree as ET
    import io
    ns = "http://jasperreports.sourceforge.net/jasperreports"
    ET.register_namespace('', ns)
    # Set each field name's width to a default value
    default_col_width = 120
    width_per_col = default_col_width
    total_width = max(595, width_per_col * len(df.columns))
    left_margin = 20
    right_margin = 20
    column_width = total_width - left_margin - right_margin
    root = ET.Element("jasperReport", attrib={
        "xmlns": ns,
        "name": report_name,
        "pageWidth": str(total_width),
        "pageHeight": "842",
        "columnWidth": str(column_width),
        "leftMargin": str(left_margin),
        "rightMargin": str(right_margin),
        "topMargin": "20",
        "bottomMargin": "20",
        "language": "plsql"
    })
    # Add default data adapter property
    ET.SubElement(root, "property", attrib={
        "name": "com.jaspersoft.studio.data.defaultdataadapter",
        "value": "Multifonds"
    })

    # Add queryString with PL/SQL call
    query_string = ET.SubElement(root, "queryString", attrib={"language": "plsql"})
    proc_call = f"{{call SP_RPT_{report_name.upper()}($P{{ASATDATE}}, $P{{PRODUCTCODE}}, $P{{ORACLE_REF_CURSOR}})}}"
    query_string.text = proc_call
    # Add <field> elements for data binding
    for col in df.columns:
        ET.SubElement(root, "field", attrib={
            "name": str(col),
            "class": "java.lang.String"
        })

    # Title band
    title = ET.SubElement(root, "title")
    title_band = ET.SubElement(title, "band", attrib={"height": "50"})
    title_text = ET.SubElement(title_band, "staticText")
    ET.SubElement(title_text, "reportElement", attrib={"x": "0", "y": "0", "width": str(column_width), "height": "50"})
    ET.SubElement(title_text, "textElement")
    title_text_value = ET.SubElement(title_text, "text")
    title_text_value.text = report_name

    # Page header band
    page_header = ET.SubElement(root, "pageHeader")
    page_header_band = ET.SubElement(page_header, "band", attrib={"height": "30"})

    # Column header band with <staticText> for each field name
    column_header = ET.SubElement(root, "columnHeader")
    band = ET.SubElement(column_header, "band", attrib={"height": "30"})
    x = 0
    for col in df.columns:
        static_text = ET.SubElement(band, "staticText")
        ET.SubElement(static_text, "reportElement", attrib={"x": str(x), "y": "0", "width": str(width_per_col), "height": "30"})
        ET.SubElement(static_text, "textElement")
        text = ET.SubElement(static_text, "text")
        text.text = str(col)
        x += width_per_col

    # Detail band
    detail = ET.SubElement(root, "detail")
    detail_band = ET.SubElement(detail, "band", attrib={"height": "30"})
    x = 0
    for col in df.columns:
        text_field = ET.SubElement(detail_band, "textField")
        ET.SubElement(text_field, "reportElement", attrib={"x": str(x), "y": "0", "width": str(width_per_col), "height": "30"})
        ET.SubElement(text_field, "textElement")
        text_field_expr = ET.SubElement(text_field, "textFieldExpression")
        text_field_expr.text = f"$F{{{col}}}"
        x += width_per_col

    # Column footer band
    column_footer = ET.SubElement(root, "columnFooter")
    column_footer_band = ET.SubElement(column_footer, "band", attrib={"height": "30"})

    # Page footer band
    page_footer = ET.SubElement(root, "pageFooter")
    page_footer_band = ET.SubElement(page_footer, "band", attrib={"height": "30"})

    # Summary band
    summary = ET.SubElement(root, "summary")
    summary_band = ET.SubElement(summary, "band", attrib={"height": "30"})

    buf = io.BytesIO()
    tree = ET.ElementTree(root)
    tree.write(buf, encoding="utf-8", xml_declaration=True)
    buf.seek(0)
    return buf

);

    p_ProductId IN VARCHAR2,

    )IS 
    BEGIN 
    OPEN p_result FOR 
 
SELECT

    {field_list}

  FROM VW_TABLENAME;
  {create_table_sql}END {proc_name};
"""
